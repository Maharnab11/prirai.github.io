    ep = 0.0001
    while True:
        if type(M1[0]) == type(M1[i]):
            m1 = M1[i]
            i+=1
            break
        i+=1
    while True and i < len(M1):
        if type(M1[0]) == type(M1[i]):
            m2 = M1[i]
            break
        i+=1
